function(par, a) {
  a = a[with(a, order(start)), ]
  nll = 0
  if (par[1] < 0 | par[1] > 1 |
      par[2] < 0 | par[2] > 50 | 
      par[3] < 0 | par[3] > 1) {
    nll = Inf
  } else {
    tempQ = c()
    Q = c(0, 0)
    date = rep(a$start[1], 2)
    t = c(0, 0)
    P <- vector()
    rewards = a$dooropened
    sides = ceiling(a$corner/2)
    nows.start = a$start
    nows.end = a$end
    for (i in seq_along(sides)) {
      r = rewards[i]
      s = sides[i]
      now.start = nows.start[i]
      now.end = nows.end[i]
      t = as.numeric(difftime(now.start, date, units = 'mins'))
      t = ifelse(t > 660, 660, t)
      date[s] = now.end
      Q[s] = exp( -(t[s]) * par[3] ) * Q[s]
      tempQ = exp( -(t[-s]) * par[3] ) * Q[-s]
      P[s] = exp(par[2] * Q[s]) / 
        (sum(exp(par[2] * Q[s]), exp(par[2] * tempQ)))
      if(P[s] < .001){P[s] = .001}
      if(P[s] > .999){P[s] = .999}
      nll = -log(P[s]) + nll
      pe = r - Q[s]
      Q[s] = Q[s] + (par[1] * pe)}}
  nll}

fn <- function(t, x = value, par = 1) exp(-t * par) * x
value = 1
temp <- runif(100, 1/60, 20/60)
check = fn(sum(temp))
for(i in temp){value = fn(i)}
param = 1

# difference
check - value
## [1] 6.617445e-24
# probability
exp(check * param) / sum(exp(check * param), exp(value * param))
## [1] 0.5
